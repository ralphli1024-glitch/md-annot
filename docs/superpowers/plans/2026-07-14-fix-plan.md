## 计划元数据

- **目标：** 按修复计划完成 6 项改动（移除尾标、工具栏优化、手写持久化、Undo/Redo、批注列表优化）
- **技能引用：** tech_loop_coder（TypeScript + Vitest 版）
- **范围边界（不在此计划内）：**
  - 不修改 `.annotations/` 目录结构或索引逻辑
  - 不增加新功能（如云端同步、PDF 导出）
  - 不重构现有模块（如 Canvas 渲染重写、存储引擎替换）
  - 不修改编辑模式浮动工具栏（floating-toolbar.ts）的功能逻辑
  - 不修改 CM6 装饰器的 changeTracker 机制
- **关键假设（待确认）：** Undo/Redo 栈仅存于内存，不持久化到磁盘

## 架构

6 项改动分布在 9 个源文件 + 1 个样式文件中，各自独立可并行。核心改动集中在：
- **UI 层**（`side-panel.ts`, `reading-toolbar.ts`, `cm6-decorations.ts`, `styles.css`）：纯样式和 DOM 调整
- **数据层**（`storage.ts`, `data-models.ts`, `canvas-overlay.ts`）：手写笔触持久化
- **状态层**（`annotation-manager.ts`, `main.ts`, `canvas-overlay.ts`）：Undo/Redo 栈集成

## 技术栈

- TypeScript 5.3 (target ES6)
- Vitest 1.0（测试框架，命令：`npx vitest run --reporter=verbose`）
- esbuild（构建：`node esbuild.config.mjs`）

- [ ] **Step2：在 `annotation-manager.ts` 中实现 UndoManager**

在 `AnnotationManager` 类中追加内部字段和方法：

```typescript
import { UndoAction } from "./data-models";  // import 中追加

export class AnnotationManager {
  // 在现有字段后追加
  private undoStack: UndoAction[] = [];
  private redoStack: UndoAction[] = [];
  private readonly maxUndoStack = 50;

  // ── CRUD 方法修改 ──

  // createAnnotation 末尾追加 undo 记录（在 emit 之前）
  createAnnotation(...): Annotation | null {
    // ... 现有逻辑 ...
    this.annotations.set(annotation.id, annotation);
    
    // 记录撤销动作：删除该批注
    this.undoStack.push({
      type: 'annotation:created',
      timestamp: Date.now(),
      payload: { id: annotation.id },
    });
    this.redoStack = []; // 新操作清空 redo
    this.trimUndoStack();

    this.emit("annotation:created", annotation);
    return annotation;
  }

  // deleteAnnotation 修改
  deleteAnnotation(id: string): boolean {
    const oldAnnotation = this.annotations.get(id);
    if (!oldAnnotation) return false;
    
    this.annotations.delete(id);
    
    // 记录撤销动作：恢复该批注
    this.undoStack.push({
      type: 'annotation:deleted',
      timestamp: Date.now(),
      payload: { ...oldAnnotation },
    });
    this.redoStack = [];
    this.trimUndoStack();

    this.emit("annotation:deleted", { id });
    return true;
  }

  // ── Undo/Redo 方法 ──

  undo(): void {
    const action = this.undoStack.pop();
    if (!action) return;

    switch (action.type) {
      case 'annotation:created': {
        // 撤销创建 = 删除批注
        const { id } = action.payload;
        const deleted = this.annotations.get(id);
        if (deleted) {
          this.annotations.delete(id);
          this.redoStack.push({
            type: 'annotation:deleted',
            timestamp: Date.now(),
            payload: { ...deleted }, // 保存完整对象用于 redo
          });
          this.emit("annotation:deleted", { id });
        }
        break;
      }
      case 'annotation:deleted': {
        // 撤销删除 = 恢复批注
        const restored = deserializeAnnotation(action.payload);
        this.annotations.set(restored.id, restored);
        this.redoStack.push({
          type: 'annotation:created',
          timestamp: Date.now(),
          payload: { id: restored.id },
        });
        this.emit("annotation:created", restored);
        break;
      }
    }
    this.emit("annotations:reanchored", Array.from(this.annotations.values()));
  }

  redo(): void {
    const action = this.redoStack.pop();
    if (!action) return;

    switch (action.type) {
      case 'annotation:created': {
        // 重做删除 = 删除批注
        const { id } = action.payload;
        const deleted = this.annotations.get(id);
        if (deleted) {
          this.annotations.delete(id);
          this.undoStack.push({
            type: 'annotation:created',
            timestamp: Date.now(),
            payload: { id },
          });
          this.emit("annotation:deleted", { id });
        }
        break;
      }
      case 'annotation:deleted': {
        // 重做恢复 = 恢复批注
        const restored = deserializeAnnotation(action.payload);
        this.annotations.set(restored.id, restored);
        this.undoStack.push({
          type: 'annotation:deleted',
          timestamp: Date.now(),
          payload: { ...restored },
        });
        this.emit("annotation:created", restored);
        break;
      }
    }
    this.emit("annotations:reanchored", Array.from(this.annotations.values()));
  }

  private trimUndoStack(): void {
    if (this.undoStack.length > this.maxUndoStack) {
      this.undoStack.shift();
    }
  }

  clearUndoStack(): void {
    this.undoStack = [];
    this.redoStack = [];
  }
  
  // clear() 方法中追加清空栈
  clear(): void {
    this.annotations.clear();
    this.currentFilePath = "";
    this.undoStack = [];
    this.redoStack = [];
  }
}
```

```typescript
      const isMod = e.metaKey || e.ctrlKey;
      if (!isMod || e.key !== "z") return;

      const activeView = this.app.workspace.getActiveViewOfType(MarkdownView);
      if (!activeView || !this.annotationManager.getCurrentFilePath()) return;

        if (strokes.length > 0) {

          this.canvasOverlay.undoLastStroke();

        e.preventDefault();
        this.annotationManager.undo();
      } else {

        this.annotationManager.redo();
```
